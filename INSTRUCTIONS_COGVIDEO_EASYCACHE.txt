if not hasattr(cls, "_original_forward"):
    cls._original_forward = cls.forward

        cls.cnt = 0
        cls.num_steps = steps
        cls.total_time = 0.0
        cls.k = None
        cls.previous_raw_input = None
        cls.previous_output = None
        cls.prev_prev_raw_input = None

        cls.pred_change_history = []

                "mode": mode_name,

                "thresh": float(cls.thresh),

                "video_path": str(out_path),

Policy details (ret_steps + thresholds)
--------------------------------------

EasyCache makes a **compute vs skip** decision at each diffusion step \(i\).

- **`ret_steps`**: the first `ret_steps` steps **always compute** (no skipping).
- **last step**: the final step \(i == num_steps-1\) **always computes**.
- **fixed threshold**: uses a single `thresh` for all non-ret/non-final steps.
- **adaptive threshold**: uses `thresh_low` near the “volatile” edges and `thresh_high` in the stable middle:

  - **low-threshold region** (more conservative): \(i < first_steps\) OR \(i >= num_steps - last_steps\)
  - **high-threshold region** (more aggressive): otherwise (middle steps)

Concrete values used in our CogVideo runs:

- **Fixed-threshold modes** (all use `ret_steps=1`):
  - `cog_ec_fixed_0.035`: `thresh=0.035`, `ret_steps=1`
  - `cog_ec_fixed_0.05`:  `thresh=0.05`,  `ret_steps=1`
  - `cog_ec_fixed_0.08`:  `thresh=0.08`,  `ret_steps=1`
  - `cog_ec_fixed_0.10`:  `thresh=0.10`,  `ret_steps=1`
  - `cog_ec_fixed_0.12`:  `thresh=0.12`,  `ret_steps=1`
  - `cog_ec_fixed_0.15`:  `thresh=0.15`,  `ret_steps=1`

- **Adaptive modes** (threshold switches by `first_steps`/`last_steps` as described above):
  - `cog_ec_adapt_lo035_hi080`:
    - `ret_steps=3`
    - `thresh_low=0.035`, `thresh_high=0.08`
    - `first_steps=10`, `last_steps=5`
  - `cog_ec_adapt_lo050_hi080`:
    - `ret_steps=2`
    - `thresh_low=0.05`, `thresh_high=0.08`
    - `first_steps=8`, `last_steps=5`
  - `cog_ec_adapt_lo050_hi100`:
    - `ret_steps=1`
    - `thresh_low=0.05`, `thresh_high=0.10`
    - `first_steps=8`, `last_steps=5`

Example adaptive run (single video) — edit the step ranges + thresholds:

```bash
cd /nfs/oagrawal/CogVideo

python3 - << 'PY'
import os, time
from dataclasses import dataclass

import torch
from diffusers import CogVideoXPipeline
from diffusers.utils import export_to_video

from easycache_cogvideo import _extract_sample_from_output, _wrap_sample_for_output

@dataclass
class AdaptiveCfg:
    ret_steps: int = 5
    thresh_low: float = 0.025
    thresh_high: float = 0.050
    first_steps: int = 12
    last_steps: int = 6

def adaptive_forward(self, hidden_states, encoder_hidden_states, timestep, image_rotary_emb=None, attention_kwargs=None, return_dict=True):
    torch.cuda.synchronize()
    start = time.time()
    raw_input = hidden_states
    steps = int(self.num_steps)
    i = int(self.cnt)

    # decide which threshold applies
    current_thresh = self.thresh_high
    if i < self.first_steps or i >= steps - self.last_steps:
        current_thresh = self.thresh_low

    if i < self.ret_steps or i >= steps - 1:
        should_calc = True
        self.accumulated_error = 0.0
    else:
        if self.previous_raw_input is not None and self.previous_output is not None and self.k is not None:
            raw_input_change = (raw_input - self.previous_raw_input).abs().mean()
            output_norm = self.previous_output.abs().mean()
            pred_change = (self.k * (raw_input_change / output_norm)).item() if output_norm > 0 else float("inf")
            self.accumulated_error += pred_change
            should_calc = self.accumulated_error >= current_thresh
            if should_calc:
                self.accumulated_error = 0.0
        else:
            should_calc = True

    self.previous_raw_input = raw_input.detach()

    if (not should_calc) and (self.cache is not None):
        result = raw_input + self.cache
        self.cnt += 1
        torch.cuda.synchronize()
        self.total_time += (time.time() - start)
        if return_dict:
            return {"sample": result}
        return (result,)

    out = self._original_forward(
        hidden_states=hidden_states,
        encoder_hidden_states=encoder_hidden_states,
        timestep=timestep,
        image_rotary_emb=image_rotary_emb,
        attention_kwargs=attention_kwargs,
        return_dict=return_dict,
    )
    result = _extract_sample_from_output(out, return_dict)

    if self.previous_output is not None and self.prev_prev_raw_input is not None:
        output_change = (result - self.previous_output).abs().mean()
        input_change = (self.previous_raw_input - self.prev_prev_raw_input).abs().mean()
        if input_change > 0:
            self.k = output_change / input_change

    self.cache = result - raw_input
    self.prev_prev_raw_input = self.previous_raw_input
    self.previous_output = result.detach()

    self.cnt += 1
    torch.cuda.synchronize()
    self.total_time += (time.time() - start)
    return _wrap_sample_for_output(result, out, return_dict)

cfg = AdaptiveCfg()
prompt = "A panda, dressed in a small red jacket and tiny hat, plays a guitar in a bamboo forest."
model_id = os.environ.get("COG_MODEL", "THUDM/CogVideoX-2b")
steps = int(os.environ.get("COG_STEPS", "50"))
guidance = float(os.environ.get("COG_GUIDANCE", "6.0"))
seed = int(os.environ.get("COG_SEED", "12345"))
out = os.environ.get("COG_OUT", "/nfs/oagrawal/CogVideo/vbench_eval_easycache/videos/cog_ec_adaptive_single.mp4")

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
g = torch.Generator(device=device).manual_seed(seed)

pipe = CogVideoXPipeline.from_pretrained(
    model_id,
    torch_dtype=torch.float16 if device.type == "cuda" else torch.float32,
).to(device)

tr = pipe.transformer
cls = tr.__class__
if not hasattr(cls, "_original_forward"):
    cls._original_forward = cls.forward

# attach adaptive forward + state
cls.forward = adaptive_forward
cls.cnt = 0
cls.num_steps = steps
cls.total_time = 0.0
cls.k = None
cls.previous_raw_input = None
cls.previous_output = None
cls.prev_prev_raw_input = None
cls.cache = None
cls.accumulated_error = 0.0
cls.ret_steps = cfg.ret_steps
cls.thresh_low = cfg.thresh_low
cls.thresh_high = cfg.thresh_high
cls.first_steps = cfg.first_steps
cls.last_steps = cfg.last_steps

t0 = time.time()
vid = pipe(prompt=prompt, num_inference_steps=steps, guidance_scale=guidance, generator=g, output_type="pt").frames[0]
e2e = time.time() - t0
export_to_video(vid, out, fps=8)
print("saved:", out, "e2e_seconds:", round(e2e,2), "transformer_seconds:", round(float(tr.total_time),2))
PY
```

================================================================================
Notes / troubleshooting from CogVideo EasyCache runs
================================================================================

- **Transformers version for VBench**
  - If you ever install or upgrade `diffusers`, `accelerate`, etc. and `transformers`
    gets bumped (e.g., to 5.x), VBench's `scene` (and possibly other) dimensions can
    break with errors like:
    - `BertTokenizer has no attribute additional_special_tokens_ids`
  - **Fix**: inside the `hv_eval_wan` container, re-pin transformers:
    - `pip install transformers==4.33.2`

- **PyTorch 2.6 `torch.load` change and motion_smoothness**
  - Newer PyTorch defaults `torch.load(..., weights_only=True)`, which can cause
    `motion_smoothness` to fail when loading the AMT checkpoint, with errors about
    `Weights only load failed` / `typing.OrderedDict` not being allowlisted.
  - **Fix**: this is already handled inside
    `HunyuanVideo/vbench_eval_easycache/run_vbench_eval.py` via a small
    `_patch_torch_load()` helper, so you do **not** need to do anything extra here.
    Just make sure you are using the updated script from `HunyuanVideo`.

- **Verifying all 16 VBench dimensions per mode**
  - After VBench finishes, you can quickly sanity check that each mode has all
    16 dimensions evaluated:
    ```bash
    COG_EC=/nfs/oagrawal/CogVideo/vbench_eval_easycache
    for mode in cog_ec_baseline cog_ec_fixed_0.035 cog_ec_fixed_0.05 \
                cog_ec_fixed_0.08 cog_ec_fixed_0.10 cog_ec_fixed_0.12 cog_ec_fixed_0.15; do
      dir="$COG_EC/vbench_scores/$mode"
      count=$(ls "$dir"/*_eval_results.json 2>/dev/null | wc -l)
      echo "$mode: $count/16 eval_results.json"
    done
    ```
